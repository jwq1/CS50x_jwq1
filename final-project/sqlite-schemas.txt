CREATE TRIGGER add_research_title AFTER INSERT ON research FOR EACH ROW BEGIN INSERT INTO research (title) VALUES (research.link); END;


CREATE TABLE research (
    id INTEGER PRIMARY KEY,
    product_id INTEGER NOT NULL,
    link TEXT NOT NULL,
    title TEXT
    );

CREATE TRIGGER add_research_title
    AFTER INSERT ON research
    FOR EACH ROW
    WHEN article_title ISNULL
BEGIN

    SET article_title=NEW.link;
END;
